Sublime Text 4

mkdir ~/Downloads
cd ~/Downloads

# 下载 SublimeText 4
wget https://download.sublimetext.com/sublime-text_build-4169_amd64.deb

# 安装 SublimeText 4
sudo dpkg -i ~/Downloads/sublime-text_build-4169_amd64.deb
